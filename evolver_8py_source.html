220 class TwoPiJumps(Evolver):
221  """!
222  Shift configuration by 2*pi at random lattice sites.
223 
224  Evolver that selects lattice sites and shifts the configuration by
225  `+2*pi` or `-2*pi` at those sites where the sign is chosen randomly per site.
226  The conjugate momentum is not changed.
227  Can modify multiple sites at the same time (one "batch") and do multiple batches.
228  Each batch is accepted or rejected on its own.
229 
230  If the action consists only of isle.action.HubbardGaugeAction and any variant
231  of isle.action.HubbardFermiAction, a shortcurt can be used to compute the change
232  in action which allows for very fast execution.
233  """
234 
235  def __init__(self, nBatches, batchSize, action, lattice, rng):
236  r"""!
237  \param nBatches Number of batches of sites.
238  \param batchSize Number of sites in each batch.
239  \param action Instance of isle.Action to use for molecular dynamics.
240  \param lattice Lattice the simulation runs on.
241  \param rng Central random number generator for the run.
242  """
243 
244  self.nBatches = nBatches
245  self.batchSize = batchSize
246  self.rng = rng
247  self.latSize = lattice.lattSize()
248  self.selector = BinarySelector(rng)
249 
250  # _sumHalfInvUtildes is either sum(1/Utilde)/2 from all HubbardGaugeActions
251  # or None
252  # _action is either None or action that was passed in (same order as _sumInvUtildes)
253  utildes = self._findUtildes(action)
254  if utildes is None:
255  self._action = action
256  self._sumHalfInvUtilde = None
257  getLogger(__name__).info("Action does not allow for evaluation shortcut under "
258  "jumps by 2pi, using normal (slow) evaluation")
259  else:
260  self._action = None
261  self._sumHalfInvUtilde = sum(1 / utilde for utilde in utildes) / 2
262  getLogger(__name__).info("Action allows for evaluation shortcut under jumps "
263  "by 2pi, found utilde = %s", utildes)
264 
265  def _findUtildes(self, action):
266  r"""!
267  Find all HubbardGaugeActions and extract Utilde.
268  \returns Either a list of all Utildes found or None if there is any action
269  which does not allow for the shortcut calculation.
270  """
271 
272  # all sub actions must allow for the shortcut
273  if isinstance(action, isle.action.SumAction):
274  utildes = []
275  for i in range(len(action)):
276  aux = self._findUtildes(action[i])
277  if aux is None:
278  return None
279  utildes.extend(aux)
280  return utildes
281 
282  # single action
283  if isinstance(action, isle.action.HubbardGaugeAction):
284  return [action.utilde]
285 
286  # action is invariant under 2pi jump
287  if isinstance(action, (isle.action.HubbardFermiActionDiaOneOne,
288  isle.action.HubbardFermiActionDiaTwoOne,
289  isle.action.HubbardFermiActionExpOneOne,
290  isle.action.HubbardFermiActionExpTwoOne)):
291  return []
292 
293  return None

295  def _evalAction(self, newPhi, sites, shifts, actVal):
296  """!
297  Compute the value of the action at a given phi after a jump.
298  """
299 
300  if self._action is None:
301  # shortcut: Computes (newPhi**2 - oldPhi**2) / (2*Utilde), the difference in
302  # gauge action. The fermion action is invariant under jumps by 2*pi
303  # if this branch is taken.
304  return actVal + np.dot(2*np.array(newPhi, copy=False)[sites]-shifts, shifts) \
305  * self._sumHalfInvUtilde
306 
307  # Some part of the action does not allow for the shortcut.
308  return self._action.eval(newPhi)

310  def evolve(self, phi, pi, actVal, trajPoint):
311  r"""!
312  Evolve a configuration, momentum remains unchanged.
313  \param phi Input configuration.
314  \param pi Input Momentum.
315  \param actVal Value of the action at phi.
316  \param trajPoint \e ignored.
317  \returns In order:
318  - New configuration
319  - New momentum
320  - Action evaluated at new configuration
321  - 0 if no jump was accepted, 1 if at least one was accepted (across all batches).
322  """
323 
324  # nothing accepted yet
325  trajPoint = 0
326 
327  for _ in range(self.nBatches):
328  # lattice sites at which to jump
329  sites = self.rng.choice(self.latSize, self.batchSize, replace=False)
330  # shifts for the above sites
331  shifts = self.rng.choice((-2*np.pi, +2*np.pi), self.batchSize, replace=True)
332 
333  # perform jumps
334  newPhi = np.array(phi, copy=True)
335  newPhi[sites] += shifts
336  newActVal = self._evalAction(newPhi, sites, shifts, actVal)
337 
338  # no need to put pi into check, it never changes in here
339  if self.selector.selectTrajPoint(actVal, newActVal) == 1:
340  phi, actVal = newPhi, newActVal
341  trajPoint = 1 # something has been accepted
342  # else:
343  # nothing to do, variables are already set up
344 
345  return phi, pi, actVal, trajPoint

371 class Alternator(Evolver):
372  r"""! \ingroup evolvers
373  Alternate between different evolvers.
374 
375  Each sub evolver is run a given number of times.
376  Then Alternator advances to the next evolver and runs it for its
377  number of iterations.
378  After the last sub evolver, Alternator cycles round to the first one and repeats.
379 
380  <B>Example</B><br>
381  This example defines an %Alternator which uses ConstStepLeapfrog for 100 trajectories,
382  then TwoPiJumps once, and repeats.
383  ```{.py}
384  # Set up parameters
385  # hmcState, length, nstep
386  # ...
387 
388  evolver = isle.evolver.Alternator((
389  (100, isle.evolver.ConstStepLeapfrog(hmcState.action, length, nstep, hmcState.rng)),
390  (1, isle.evolver.TwoPiJumps(hmcState.lattice.lattSize(), 1, hmcState.action,
391  hmcState.lattice, hmcState.rng))
392  ))
393  ```
394  """

604 def saveEvolverType(evolver, h5group, definitions={}):
605  r"""! \ingroup evolvers
606  Save an evolver's type to HDF5.
607 
608  There are three possible scenarios:
609  - The evolver is built into Isle: Only its name is stored.
610  It can be reconstructed automatically.
611  - The evolver is custom defined and included in parameter `definitions`:
612  Only its name is stored. It needs to be passed to loadEvolverType() when
613  reconstructing.
614  - The evolver is custom defined and not included in `definitions`:
615  The full source code of the evolver's definition is stored.
616  This requries that the evolver is fully self-contained, i.e. not use
617  any symbols from outside its own definition (except for `isle`).
618 
619  \see loadEvolverType to load evolvers saved with this function.
620  """
621 
622  if not isinstance(evolver, Evolver):
623  getLogger(__name__).error("Can only save instances of subclasses of "
624  "isle.evolver.Evolver, given %s",
625  type(evolver))
626  raise ValueError("Not an evolver")
627 
628  # get the name of the evolver's class
629  name = type(evolver).__name__
630  if name == "__as_source__":
631  getLogger(__name__).error("Evolvers must not be called __as_source__. "
632  "That name is required for internal use.")
633  raise ValueError("Evolver must not be called __as_source__")
634 
635  if evolver.__module__ == "isle.evolver" or type(evolver).__name__ in definitions:
636  # builtin or custom
637  h5group["__name__"] = name
638  getLogger(__name__).info("Saved type of evolver %s via its name", name)
639 
640  else:
641  # store source
642  h5group["__name__"] = "__as_source__"
643  src = sourceOfClass(type(evolver))
644  # attempt to reconstruct it to check for errors early
645  import isle
