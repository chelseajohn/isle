182 def continueRun(infile, outfile, startIdx, overwrite, definitions={}):
183  r"""!
184  Continue a previous %HMC run.
185 
186  Loads metadata and a given checkpoint from the input file, constructs a new HMC driver
187  object from them, and initializes the output file.
188 
189  \param infile Name of the input file. Must contain at least one checkpoint to continue from.
190  \param outfile Name of the output file. Can be `None`, which means equal to the input file.
191  \param startIdx Index of the checkpoint to start from. See parameter `overwrite`.
192  Can be negative in which case it is counted from the end (-1 is last checkpoint).
193  The number the checkpoint is saved as, not 'the nth checkpoint in the file'.
194  \param overwrite If `False`, nothing in the output file will be erased/overwritten.
195  If `True`,
196  - (`infile==outfile`): all configurations and checkpoints newer than `startIdx`
197  are removed and have to be re-computed.
198  - (`infile!=outfile`): outfile is removed and re-initialized,
199  whereby all content is lost.
200  \param definitions Dictionary of mapping names to custom types. Used to control how evolvers
201  are stored for checkpoints. See evolvers.saveEvolverType().
202 
203  \returns In order:
204  - Instance of HMC constructed from parameters found in `infile`.
205  - Configuration loaded from checkpoint.
206  - Evolver loaded from checkpoint.
207  - Save frequency computed on last two configurations.
208  `None` if there is only one configuration.
209  - Checkpoint frequency computed on last two checkpoints.
210  `None` if there is only one checkpoint.
211  """
212 
213  if infile is None:
214  getLogger(__name__).error("No input file given for HMC driver in continuation run")
215  raise ValueError("No input file")

217  getLogger(__name__).info("No output file given for HMC driver")
218  outfile = infile
219
