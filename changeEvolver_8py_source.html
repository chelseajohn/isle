14 def _buildEvolver(previous, hmcState):
15  """Construct an evolver for production based on the one used for thermalization."""
16 
17  # Try to read the parameters for leapfrog from the old proposer.
18  if isinstance(previous, isle.evolver.LinearStepLeapfrog):
19  # There is no way to get the current values of there paramters from the evolver.
20  # So assume it has finished and just use last values for trajectory length
21  # and number of MD steps.
22  length = previous.lengthRange[1]
23  nstep = previous.nstepRange[1]
24 
25  elif isinstance(previous, isle.evolver.ConstStepLeapfrog):
26  # Copy the exact parameters.
27  length = previous.length
28  nstep = previous.nstep
29 
30  else:
31  # Cannot handle any evolver that is not one of the above.
32  getLogger(__name__).error("Unable to extract leapfrog parameters from evolver of type %s",
33  type(previous))
34  raise RuntimeError("Unable to extract leapfrog parameters")
35 
36  # Define a new evolver that alternates between leapfrog integration and
37  # jumps py 2*pi in the field variables.
38  return isle.evolver.Alternator((
39  # Use leapfrog with constant parameters for 100 trajectories.
40  (100, isle.evolver.ConstStepLeapfrog(hmcState.action, length, nstep, hmcState.rng)),
41  # Use 2*pi jumps once but attempt to jump nt*nx times on random sites,
42  # accepting or rejecting each jump individually.
43  (1, isle.evolver.TwoPiJumps(hmcState.lattice.lattSize(), 1, hmcState.action,
44  hmcState.lattice, hmcState.rng))
