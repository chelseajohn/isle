53 class ETA:
54  """!
55  Estimate the time of arrival for iterations.
56 
57  The ETA is computed from a linear regression to the starting time
58  and current time (time at execution of the __call__ method).
59  This is not very stable for strongly changing durations of individual iterations
60  but gives a good estimate for mostly stable durations.
61 
62  The start time is only set after the first iteration
63  (calling __call__ with current > 0).
64  This is because the first iteration might perform some expensive setup
65  operations the following iterations do not have to repeat.
66  In such a case, using the time at current=0 would introduce a bias
67  to the ETA.
68 
69  Given an initial iteration xi which was started at time ti
70  and a current iteration xc and current time tc
71  the estimated final time (ETA) is
72  tf = mc * (xf - xi) + ti
73  where
74  mc = (tc - ti) / (xc - xi).
75  """
76 
77  def __init__(self, target):
78  """!Initialize with a given target iteration number."""
79 
80  if target <= 1:
81  raise ValueError(f"Target iteration of ETA must be > 1, got {target}")
82 
83  self.targetIteration = target
84  self._ti = None # initial time
85  self._xi = None # iteration when initial time was measured
86 
87  def __call__(self, current):
88  r"""!
89  Estimate the time of arrival given a current iteration.
90  \param current Iteration number the loop is currently at.
91  \returns - Estimated time of arrival in seconds since epoch.
92  - `None` if no starting time has been set yet or `current`
93  is below or equal to starting iteration.
94  """
95 
96  # can't estimate time yet
97  if self._ti is None:
98  # initial time is time after first iteration or later
99  if current > 0:
100  self._ti = time.time()
101  self._xi = current
102  return None
103 
104  # The method might be called multiple times during the same iteration.
105  # But cannot estimate while still at initial iteration (xi).
106  if current <= self._xi:
107  return None
108 
109  # do linear regression
110  tc = time.time()
111  return (tc-self._ti)/(current-self._xi) * (self.targetIteration-self._xi) + self._ti
112

495 @contextlib.contextmanager
496 def trackProgress(target, message="", updateRate=1):
497  r"""!
498  A context manager to track progress of an operation via a progress bar.
499 
500  Sets up and returns a new progress bar.
501  The caller needs to advance that bar themselves.
502 
503  \param target Target number of steps to track.
504  Can be `None` in which case the bar only indicates that something,
505  happens, not how far away the goal is.
506  \param message Message to display in front of the progress bar.
507  \returns A newly constructed instance of Progressbar.
508  \throws RuntimeError is a progress bar is already active.
509  """
510 
511  global _activeBar
512  if _activeBar is not None:
513  logging.getLogger(__name__).error("Cannot construct a new progress bar, "
514  "another one is already active.")
515  raise RuntimeError("A progress bar is already active.")
516 
517  try:
518  _activeBar = makeProgressbar(target, message, updateRate)
519  yield _activeBar
520  _activeBar.finalize() # success => clean up
521 
522  except:
523  # failure => leave bar visible and advance a line
524  sys.stderr.write("\n")
525  raise
526 
527  finally:
528  # in any case the bar is now done
529  _activeBar = None

531 def progressRange(start, stop=None, step=1, message="", updateRate=1):
532  r"""!
533  Like built in `range()` but indicates progress using a Progressbar.
534 
535  Parameters `start`, `stop`, and `step` behave the same way as for
536  the built in `range()` generator.
537  `message` is displayed in front of the progress bar.
538  """
539 
540  # mimic behavior of built in range
541  if stop is None:
542  stop = start
543  start = 0
544 
545  with trackProgress(stop-start, message, updateRate) as pbar:
546  for cur in range(start, stop, step):

549  # If stop-start is not a multiple of step, advance(step) would overshoot.
550  pbar.advance(min(step, stop-cur))
