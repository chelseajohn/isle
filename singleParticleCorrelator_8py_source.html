34  # Create a large set of sources:
35  rhss = [isle.Vector(spaceToSpacetime(irrep, time, nt))
36  for irrep in self.irreps for time in range(nt)]
37  # For the j^th spacetime vector of the i^th state, go to self.rhss[i * nt + j]
38  # In other words, time is the faster running index.
39 
40  # Solve M*x = b for all right-hand sides:
41  if self._alpha == 1:
42  res = np.array(isle.solveM(self.hfm, phi, self.species, rhss), copy=False)
43  else:
44  res = np.linalg.solve(isle.Matrix(self.hfm.M(-1j*phi, self.species)), np.array(rhss).T).T
45 
46  propagators = res.reshape([len(self.irreps), nt, nt, len(self.irreps)])
47  # The logic for the one-liner goes as:
48  # For each source irrep we need to apply a sink for every irrep.
49  # This produces a big cross-correlator.
50  # For each source time we need to roll the vector such that the
51  # source lives on timeslice 0.
52  # Finally, we need to average over all the source correlators with
53  # the same source irrep but different timeslices.
54  self.corr.append(np.mean(
55  np.array([
56  [
57  [rollTemporally(np.dot(propagators[i, src], np.conj(irrepj)), -src)
58  for src in range(nt)]
59  for i in range(len(self.irreps))]
60  for irrepj in self.irreps]),
