18 def main():
19  # Initialize Isle.
20  # This sets up the command line interface, defines an argument parser for a measurement
21  # command, and parses and returns arguments.
22  args = isle.initialize("meas", prog="measure")
23 
24  # Set up a measurement driver to run the measurements.
25  measState = isle.drivers.meas.init(args.infile, args.outfile, args.overwrite)
26  # The driver has retrieved all previously stored parameters from the input file,
27  params = measState.params
28  # as well as the lattice including the number of time slices nt.
29  lat = measState.lattice
30 
31  # For simplicity do not allow the spin basis.
32  assert params.basis == isle.action.HFABasis.PARTICLE_HOLE
33 
34  # Get "tilde" parameters (xTilde = x*beta/Nt) needed to construct measurements.
35  muTilde = params.tilde("mu", lat)
36  kappaTilde = params.tilde(measState.lattice.hopping(), lat.nt())
37 
38  # This object is a lower level interface for the Hubbard fermion action
39  # needed by some measurements. The discretization (hopping) needs
40  # to be selected manually.
41  if params.hopping == isle.action.HFAHopping.DIA:
42  hfm = isle.HubbardFermiMatrixDia(kappaTilde, muTilde, params.sigmaKappa)
43  else:
44  hfm = isle.HubbardFermiMatrixExp(kappaTilde, muTilde, params.sigmaKappa)
45 
46  # Define measurements to run.
47  #
48  # The measurements are run on each configuration in the slice passed to 'configSlice'.
49  # It defaults to 'all configurations' in e.g. the Logdet measurement.
50  # The two correlator measurements are called for every 10th configuration only
51  # but across the entire range of configurations.
52  #
53  # The string parameter in each constructor call is the path in the output file
54  # where the measurement shall be stored.
55  # The driver ensures that the location can be written to and that nothing gets
56  # overwritten by accident.
57  measurements = [
58  # log(det(M)) where M is the fermion matrix
59  isle.meas.Logdet(hfm, "logdet"),
60  # \sum_i phi_i
61  isle.meas.TotalPhi("field"),
62  # copy the action into the output file
63  isle.meas.Action("action"),
64  # single particle correlator for particles / spin up
65  isle.meas.SingleParticleCorrelator(hfm, isle.Species.PARTICLE,
66  "correlation_functions/single_particle",
67  configSlice=s_[::10]),
68  # single particle correlator for holes / spin down
69  isle.meas.SingleParticleCorrelator(hfm, isle.Species.HOLE,
70  "correlation_functions/single_hole",
71  configSlice=s_[::10]),
72  ]
73 
74  # Run the measurements on all configurations in the input file.
75  # This automatically saves all results to the output file when done.
76  measState(measurements)
77 
78 if __name__ == "__main__":
79  main()
