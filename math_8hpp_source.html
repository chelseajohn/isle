270

272  const std::size_t n = matrix.rows();

274  if (n != matrix.columns())

276 #endif
277 
278  // pivot indices
279  std::unique_ptr<int[]> ipiv = std::make_unique<int[]>(n);
280  // perform LU decomposition (mat = PLU)

282

285  for (std::size_t i = 0; i < n; ++i) {
286  // determinant of pivot matrix P
287  if (ipiv[i]-1 != blaze::numeric_cast<int>(i)) {
288  negDetP = !negDetP;
289  }
290  // log det of U (diagonal elements)

292  }

294  return toFirstLogBranch(res + (negDetP ? std::complex<ET>{0, pi<ET>} : 0));
