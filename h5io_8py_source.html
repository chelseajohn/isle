177

183  contain only a subset of all actions.
184  \param base Path in HDF5 file under which the action is stored.
185  \returns (action, configRange) where

190  """

193  action = None
194

198  minComponents=3),

234 def loadActionWeightsFor(dset, base="/"):
235  r"""!
236  Load the weights from the imaginary part of the action for a measurement result.
237 
238  The weights are loaded based on the 'configurations' attribute stored in the
239  parent group of `dset`.
240  This requires the attribute to be stored properly (no `None`) and the file to
241  contain the values of the action for all the trajectories the measurement was taken on
242  (can be as '/action/action' dataset or '/configuration' group).
243 
244  The weights are computed as \f$e^{-i S_{I}}\f$.
245 
246  \param dset HDF5 dataset containing the measurement result.
247  \param base Path in HDF5 file under which the action is stored.
248  \returns np.ndarray of the weights for `dset`.
249  """
250 
251  # load the configuration slice
252  group = dset.parent
253  try:
254  neededRange = parseSlice(group.attrs["configurations"], minComponents=3)
255  except KeyError:
256  getLogger(__name__).error("Cannot load weights for dataset %s; "
257  "there is no 'configurations' attribute", dset)
258  raise
259  if None in (neededRange.start, neededRange.stop, neededRange.step):
260  getLogger(__name__).error("The 'configurations' attibute for dataset %s "
261  "has not been saved properly, it contains None: %s",
262  dset, neededRange)
263  raise RuntimeError("configSlice of dataset contains None")
264 
265  action, actionRange = loadActionValuesFrom(dset, base=base)
266  try:
267  subRange = subslice(actionRange, neededRange)
268  except ValueError:
269  # try again, maybe there are enough configurations in the file
270  action, actionRange = loadActionValuesFrom(dset, True, base=base)
271  subRange = subslice(actionRange, neededRange)
272 
273  return np.exp(-1j * np.imag(action[subRange]))
