124  r"""!
125  Construct a new class for parameters. Uses either a dataclass or custom workaround.
126  """
127  def _tilde(self, value, nt, beta=None):
128  if beta is None:
129  # try to used stored beta
130  if not hasattr(self, "beta"):
131  raise RuntimeError("No parameter 'beta' stored. "
132  "Pass it as parameter to tilde().")
133  beta = self.beta
134 
135  # normalize parameters
136  if isinstance(value, str):
137  value = getattr(self, value)
138  if isinstance(nt, Lattice):

145  return make_dataclass("Parameters",
146  ((key, type(value), field(default=value))
147  for key, value in fields.items()),
148  namespace={"asdict": asdict,

170 def parameters(**kwargs):
171  r"""!
172  Return a dataclass instance holding the parameters passed to this function.
173 
174  Can take any arbitrary keyword arguments and makes a new dataclass
175  to store them. Every time this function gets called, it creates a new
176  class and immediately constructs an instance. The returned object is read only.
177 
178  The returned object has additional methods on top of the usual
179  dataclass members:
180  - <B>asdict</B>():
181  Returns a dict mapping attribute names to values.
182  - <B>tilde</B>(value, nt, beta):
183  Return `value*beta/nt`.
184  Parameters:
185  - `value`: If `str`, read attribute with that name from dataclass,
186  else use argument directly.
187  - `nt`: Number of time slices or isle.Lattice.
188  - `beta`: Inverse temperature. Read from dataclass attribute 'beta'
189  if argument set to `None` (default).
190 
191  The new classes are automatically registered with YAML so they can
192  be dumped. A loader is registered if yamlio is imported.
193 
194  \warning Dataclasses were added in Python3.7.
195  This function uses a workaround in earlier versions.
196  That workaround has greatly reduced features compared to dataclasses.
197  Be careful when you need portability.
198  """
199 
200  cls = _makeParametersClass(kwargs)
201  yaml.add_representer(cls, lambda dumper, params:
202  dumper.represent_mapping("!parameters",
203  params.asdict(),
204  flow_style=False))
