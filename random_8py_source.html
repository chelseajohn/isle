137  r"""!
138  Write the current state of the RNG into a HDF5 group.
139  \param group ´h5.Group` instance to write into. No new group is created inside of it.
140  """
141 
142  stDat = self._state.get_state()
143  assert stDat[0] == self.NAME.rsplit(".")[-1]
144 
145  # write state
146  group["name"] = self.NAME
147  group["keys"] = stDat[1]
148  group.attrs["pos"] = stDat[2]
149  group.attrs["has_gauss"] = stDat[3]
150  group.attrs["cached_gaussian"] = stDat[4]

152  def readH5(self, group):
153  r"""!
154  Read the RNG state from HDF5.
155  \param group `h5.Group` instance which contains all needed data.
156  """
157 
158  try:
159  # make sure this is the correct RNG
160  if group["name"][()] != self.NAME:
161  raise RuntimeError("Wrong kind of RNG. Expected {}, got {}"
162  .format(self.NAME, group["name"]))
163 
164  # load state
165  self._state.set_state((self.NAME.rsplit(".")[-1],
166  np.array(group["keys"]),
167  group.attrs["pos"],
168  group.attrs["has_gauss"],
169  group.attrs["cached_gaussian"]))
170  except KeyError as err:
171  # add some info and throw it back at my face
172  err.args = ("Malformatted RNG group: {}".format(err), )
173  raise err

187 def readStateH5(group):
188  r"""!
189  Construct a new RNG state from HDF5.
190  \param group HDF5 group that contains data for an RNG state. Must contain a dataset
191  called 'name' that indicates which RNG is stored. Other fields are
192  specific to the RNG that was stored.
193  \returns A new RNG state instance. The type is determined from HDF5.
194  """
195  if "name" not in group:
196  raise ValueError("HDF5 group does not contain an RNG")
197  rng = RNGS[group["name"][()]](0) # instantiate new RNG
198  rng.readH5(group) # read state
