56 def oneDimKDE(dat, bandwidth=0.2, nsamples=1024, kernel="gaussian"):
57  """!
58  Perform a 1D kenrel density estimation on some data.
59  \returns Tuple of sampling points and densities.
60  Or return (None, None) if scikit-learn is not available.
61  """
62 
63  if _DO_KDE:
64  # make 2D array shape (len(totalPhi), 1)
65  twoDDat = np.array(dat)[:, np.newaxis]
66  # make 2D set of sampling points
67  samplePts = np.linspace(np.min(dat)*1.1, np.max(dat)*1.1, nsamples)[:, np.newaxis]
68  # estimate density
69  dens = np.exp(KernelDensity(kernel=kernel, bandwidth=bandwidth)
70  .fit(twoDDat)
71  .score_samples(samplePts))
72  return samplePts[:, 0], dens
73 
74  return None, None

76 def runningAverage(data, binsize):
77  """!
78  Compute an average over the previous `binsize` points for each point in `data[binsize:]`.
79  """
80  data = np.array(data)
81  indices = np.arange(binsize, len(data))
82  density = np.empty(len(indices), dtype=float)
83  for i, j in enumerate(indices):
84  density[i] = np.mean(data[j-binsize:j])
85  return indices, density

235 def plotCorrelators(measState, axP, axH):
236  r"""!
237  Plot particle and hole Correlators.
238  The correlators are reweighted with the imaginary part of the action if possible.
239  \returns True if successful, False if no data was found.
240  """
241 
242  # load data from previous measurement
243  with h5.File(str(measState.infile), "r") as h5f:
244  if "correlation_functions" in h5f:
245  dsetP = h5f["correlation_functions/single_particle/correlators"]
246  corrP = dsetP[()]
247  dsetH = h5f["correlation_functions/single_hole/correlators"]
248  corrH = dsetH[()]
249 
250  try:
251  weightP = isle.h5io.loadActionWeightsFor(dsetP)
252  weightH = isle.h5io.loadActionWeightsFor(dsetH)
253  except KeyError:
254  getLogger(__name__).warning("Unable to load action to do reweighting for correlators."
255  "Showing correlators without applying weights.")
256  weightP = np.ones(dsetP.shape[0])
257  weightH = np.ones(dsetH.shape[0])
258  else:
259  getLogger(__name__).error("No correlation functions found.")
260  return False
261 
262  # ensemble averages
263  corrP = np.sum(corrP*weightP.reshape(-1, 1, 1, 1), axis=0) / np.sum(weightP)
264  corrH = np.sum(corrH*weightH.reshape(-1, 1, 1, 1), axis=0) / np.sum(weightH)
265 
266  # plot all correlators
267  nx = corrP.shape[0]
268  for i, j in zip(range(nx), range(nx)):
269  axP.plot(np.real(corrP[i, j, :]), color="C0", alpha=0.6)
270  for i, j in zip(range(nx), range(nx)):
271  axH.plot(np.real(corrH[i, j, :]), color="C0", alpha=0.6)
272 
273  axP.set_title("Particles")
274  axH.set_title("Holes")
275  axP.set_xlabel(r"$N_t$")
276  axH.set_xlabel(r"$N_t$")
277  axP.set_yscale("log")
278  axH.set_yscale("log")
279
