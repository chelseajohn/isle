9 def hingeRange(start, end, stepSize):
10  r"""!
11  A generator that behaves similarly to the builtin range with two differences:
12  `%hingeRange`
13  - keeps going after reaching the end value and yields `end` forever and
14  - accepts floats as parameters.
15 
16  \param start Start value (inclusive).
17  \param end End value (exculsive if called only `int((end-start)/stepSize)` times).
18  \param stepSize Size of the steps while iterating from start to end.
19  Must be negative if `end < start`.
20  """
21 
22  cur = start # current value
23  # iterate from start to end
24  while (stepSize > 0 and cur < end) or (stepSize < 0 and cur > end):
25  yield cur
26  cur += stepSize
27  # stay at end value forever
28  while True:
29  yield end

31 def listToSlice(lst):
32  r"""!
33  Convert a list to a `slice`.
34  \param lst List of values in a uniform range.
35  \returns `slice s` such that `list(range(s.start, s.stop, s.step)) == lst`.
36  \throws ValueError if list has less than two elements or the step size is not uniform.
37  """
38 
39  if len(lst) < 2:
40  raise ValueError("List must have at least two entries to convert to slice")
41 
42  diffs = [l - lst[i-1] for i, l in enumerate(lst[1:], 1)]
43  if diffs.count(diffs[0]) != len(diffs):
44  raise ValueError("Not all step sizes are equal in list, can't convert to slice")
45 
46  return slice(lst[0], lst[-1]+diffs[0], diffs[0])

203 def subslice(large, small):
204  r"""!
205  Construct a slice for a sub collection.
206 
207  Given an *unknown* collection of data `full` and a subset of it `sub = full[large]`,
208  construct and return a slice such that
209  ```{.py}
210  sub[subslice(large, small)] == full[small]
211  ```
212 
213  \throws ValueError if `large` and `small` do not allow for such a construction,
214  e.g. small.start must be reachable from large.start in steps of size large.step.
215  """

218  raise ValueError(f"Invalid slices: large={large}, small={small}")
